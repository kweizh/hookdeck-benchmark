## Transformation Code
```js
addHandler("transform", (request, context) => {
  const obj = request.body && request.body.data && request.body.data.object;
  if (!obj) return null;
  if (obj.amount < 100) return null;  // drop low-amount events
  request.body = {
    id: obj.id,
    amount: obj.amount,
    currency: obj.currency,
    email: obj.customer_email  // renamed from customer_email
  };
  request.headers["x-hookdeck-transformed"] = "true";  // inject header
  return request;
});
```
