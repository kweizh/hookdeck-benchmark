    'name': f'secure-source-{run_id}',

    'config': {
        'auth_type': 'BASIC_AUTH',
        'auth': {
            'username': 'eval-user',
            'password': 'eval-pass'
        },
        'custom_response': {
            'content_type': 'json',
            'body': '{\"ok\": true, \"id\": \"abc\"}'

}

    url,
    data=json.dumps(payload).encode('utf-8'),
    headers={

        'Content-Type': 'application/json',
        'User-Agent': 'Mozilla/5.0'
    },
    method='POST'
)
